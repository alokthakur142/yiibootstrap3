 32:      * This method will render a link that can trigger the sorting if the column is sortable.

 34:     protected function renderHeaderCellContent()
 35:     {
 36:         if ($this->grid->enableSorting && $this->sortable && $this->name !== null) {
 37:             $sort = $this->grid->dataProvider->getSort();
 38:             $label = isset($this->header) ? $this->header : $sort->resolveLabel($this->name);
 39: 
 40:             if ($sort->resolveAttribute($this->name) !== false) {
 41:                 $label .= '<span class="caret"></span>';
 42:             }
 43: 
 44:             echo $sort->link($this->name, $label, array('class' => 'sort-link'));
 45:         } else {
 46:             if ($this->name !== null && $this->header === null) {
 47:                 if ($this->grid->dataProvider instanceof CActiveDataProvider) {
 48:                     echo CHtml::encode($this->grid->dataProvider->model->getAttributeLabel($this->name));
 49:                 } else {
 50:                     echo CHtml::encode($this->name);
 51:                 }
 52:             } else {
 53:                 parent::renderHeaderCellContent();
 54:             }
 55:         }
 56:     }

 74:     {

 78:             if ($this->filter !== false && $this->grid->filter !== null && $this->name !== null && strpos(

 84:                     $filterInputOptions = $this->filterInputOptions;

 87:                     }
 88:                 } else {
 89:                     $filterInputOptions = array();

 91:                 if (isset($filterInputOptions['class']))
